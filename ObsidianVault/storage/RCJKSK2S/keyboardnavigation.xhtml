function getCode(event) { var code = null; if (event.which == undefined) { code = event.keyCode; } else if (event.which != 0 && event.charCode != 0) { code = event.which; } if (code >= 97 && code <= 122 && event.shiftKey == true) { //caps lock is on code = code - 97 + 65; } else if (code <= 90 && code >= 65 && event.shiftKey == false) { code = code - 65 + 97; } if (code != null) { return String.fromCharCode(code); } else { return null; } } function bindCode(event, code, id) { if (getCode(event) == code) { document.getElementById(id).click(); } } function bindLink(event, code, id) { if (getCode(event) == code) { document.getElementById(id).onclick(); } }
